# IMPORTANT (engine install reality — see em-fdtd-runner/GPU_NODE_PROVISIONING.md):

#   conda env create -f gprMax/conda_env.yml && conda activate gprMax
#   cd gprMax && python setup.py build && python setup.py install
#   pip install pycuda h5py        # h5py (BSD-3) reads the .out HDF5 output

# never distributed to buyers (no copyleft distribution trigger); no AGPL deps.
